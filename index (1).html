/** * Grenada Studio — lead + application capture (with CV upload) * Google Apps Script bound to your Google Sheet. Handles: * 1. Email signups from the coming-soon page (type: "signup") * 2. Job applications from the careers pages (type: "application") * — including a CV file, saved to Google Drive with the link in the Sheet. * * SETUP * 1. Open your Google Sheet → Extensions → Apps Script. * 2. Paste this whole file in (replace Code.gs contents). Save. * 3. (Optional) change CV_FOLDER_NAME below if you want a different Drive folder. * 4. Deploy → New deployment → type "Web app": * - Execute as: Me * - Who has access: Anyone * On first deploy Google will ask you to authorize Drive + Sheets access — allow it. * 5. Copy the /exec URL. It's already wired into the site * (index.html and careers/careers.js). If the URL changes on a * redeploy, update ENDPOINT in both of those files. * * WHERE CVs GO * A folder named below is created in the Drive of the account that owns the * script. Each CV is saved as " — — .". * By default files stay PRIVATE (only you / the owner can open the link). * To let anyone with the link view, set SHARE_CV = true. */ var CV_FOLDER_NAME = 'Grenada Studio — CVs'; var SHARE_CV = false; // true = anyone with the link can view the CV var HEADERS = ['Timestamp', 'Type', 'Role', 'Name', 'Email', 'Phone', 'LinkedIn', 'CV', 'Source', 'User Agent']; function doPost(e) { var lock = LockService.getScriptLock(); lock.tryLock(30000); try { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0]; if (sheet.getLastRow() === 0) sheet.appendRow(HEADERS); var data = {}; try { data = JSON.parse(e.postData.contents); } catch (err) { data = {}; } if (!data.email && e.parameter && e.parameter.email) data = e.parameter; var email = String(data.email || '').trim(); if (!email) return json({ ok: false, error: 'no email' }); // Save CV to Drive if present var cvUrl = ''; if (data.cvData && data.cvName) { try { cvUrl = saveCv_(data); } catch (err) { cvUrl = 'ERROR: ' + err; } } sheet.appendRow([ data.ts || new Date().toISOString(), data.type || 'signup', data.role || '', data.name || '', email, data.phone || '', data.linkedin || '', cvUrl, data.source || '', data.ua || '' ]); return json({ ok: true, cv: cvUrl }); } catch (err) { return json({ ok: false, error: String(err) }); } finally { lock.releaseLock(); } } function saveCv_(data) { var folder = getFolder_(CV_FOLDER_NAME); var bytes = Utilities.base64Decode(data.cvData); var mime = data.cvMime || 'application/octet-stream'; // Build a clean filename, keeping the original extension. var orig = String(data.cvName || 'cv'); var ext = ''; var dot = orig.lastIndexOf('.'); if (dot > -1) ext = orig.substring(dot); // includes the dot var safe = function (s) { return String(s || '').replace(/[\\/:*?"<>|]+/g, ' ').trim(); }; var stamp = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyy-MM-dd HH-mm'); var name = (safe(data.name) || 'Applicant') + ' — ' + (safe(data.role) || 'Role') + ' — ' + stamp + ext; var blob = Utilities.newBlob(bytes, mime, name); var file = folder.createFile(blob); if (SHARE_CV) { try { file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); } catch (e) {} } return file.getUrl(); } function getFolder_(name) { var it = DriveApp.getFoldersByName(name); return it.hasNext() ? it.next() : DriveApp.createFolder(name); } // Open the /exec URL in a browser to confirm it's live. function doGet() { return json({ ok: true, status: 'Grenada Studio endpoint live' }); } function json(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
